Calculando KMeans para Iris¶

In [4]:
import seaborn as sns

from sklearn.cluster import KMeans

from src.metricas import calculate_wcss, optimal_number_of_clusters


# preparando nossos dados
iris = sns.load_dataset('iris')
df = iris.drop('species', axis=1)

# calculando a soma dos quadrados para as 19 quantidade de clusters
sum_of_squares = calculate_wcss(df)

# calculando a quantidade tima de clusters
n = optimal_number_of_clusters(sum_of_squares)

# rodando o kmeans para nossa quantidade tima de clusters
kmeans = KMeans(n_clusters=n)
clusters = kmeans.fit_predict(df)

import pandas as pd
from sklearn.preprocessing import LabelEncoder

import matplotlib.pyplot as plt


# transformando as especies de iris em numeros para colorir o grfico
species_encoded = LabelEncoder().fit_transform(iris['species'])

# criando um dataset contendo os clusters e as especies para usar como mapa de cores 
df_clusters = pd.concat([pd.Series(species_encoded), pd.Series(clusters)], axis=1)
df_clusters.columns = ['antes', 'depois']
